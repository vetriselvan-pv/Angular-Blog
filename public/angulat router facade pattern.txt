import { Injectable } from '@angular/core';
import { Router, NavigationExtras, ActivatedRoute } from '@angular/router';

@Injectable({
  providedIn: 'root', // Make it a singleton service
})
export class RouterFacadeService {
  constructor(private router: Router) {}

  /**
   * Navigates to a specified path.
   * @param path The array of path segments.
   * @param extras Optional configuration for the navigation.
   */
  public navigate(path: any[], extras?: NavigationExtras): Promise<boolean> {
    const finalExtras = extras.replaceUrl ? extras : {...extras, replaceUrl : true}
    return this.router.navigate(path, extras);
  }

  /**
   * Navigates to a relative path from the current route.
   * @param path The array of path segments.
   * @param route The ActivatedRoute to base the relative navigation on.
   * @param extras Optional configuration for the navigation.
   */
  public navigateRelativeTo(
    path: any[],
    route: ActivatedRoute,
    extras?: Omit<NavigationExtras, 'relativeTo'>
  ): Promise<boolean> {
    const finalExtras: NavigationExtras = {
      ...extras,
      relativeTo: route,
    };
    return this.router.navigate(path, finalExtras);
  }

  /**
   * Navigates to an absolute URL (useful for external links).
   * @param url The external URL string.
   */
  public navigateByUrl(url: string): Promise<boolean> {
    return this.router.navigateByUrl(url);
  }

  // --- Add application-specific methods here ---
 
  /**
   * Navigates to the application's dashboard.
   * if you want to hanlde based on the logged in user or some other condition also. you can implemented in this service file and call it across the application
   */
  public goToDashboard(): Promise<boolean> {
    return this.navigate(['/dashboard']);
  }
}
